1 // Here is a simple C function
 2 int add(int a, int b)
 3 {
 4         int x = 0;
 5 
 6         x = a + b;
 7 
 8         return x;
 9 }
10 
11 # Compiled MIPS code. No optimization. 
12 # Directives are removed.
13 add:
14         # allocate space 
15         addiu   $sp, $sp, -24
16 
17         # save $fp
18         sw      $fp, 20($sp)
19         # note that $fp does not point the first word.
20         # instead, it is the current $sp (pointing to the last wrod).
21         move    $fp, $sp
22 
23         # save arguments a and b onto the stack
24         # the space is allocated by the caller
25         sw      $4, 24($fp)
26         sw      $5, 28($fp)
27 
28         # x = 0   x's is located at 8($fp)
29         sw      $0, 8($fp)
30 
31         # load a and b
32         lw      $3, 24($fp)
33         lw      $2, 28($fp)
34 
35         # x = a + b 
36         addu    $2, $3, $2
37         sw      $2, 8($fp)
38 
39         # set return value
40         lw      $2, 8($fp)
41 
42         # restore stack 
43         move    $sp, $fp
44 
45         lw      $fp, 20($sp)
46         addiu   $sp, $sp, 24
47 
48         jr  $31
49         nop                     # we will learn later why nop is needed here
50 
51 # explanation of some directives
52 
53 #       .frame   framereg, framesize, returnreg
54 # argument 1: framereg is normally $sp or $fp; 
55 # argument 2: the size of the frame, and
56 # argument 3: the register that holds the return address (almost always $31).
57 #    .frame  $fp, 24, $31      # vars= 8, regs= 1/0, args= 0, gp= 8
58 
59 #       .mask    bitmask, frameoffset
60 # argument 1: 32-bit mask indicates what regiters are saved. 
61 # argument 2: offset for the location where the first register (from higher numbered) is saved.  
62 # In this example, only $fp is saved. So only bit 30 is set (0x40000000).
63 # $fp is saved at the location (framesize - 4)($fp), which is 20($fp).
64 #    .mask   0x40000000, -4
