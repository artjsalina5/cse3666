112 int     pointers()
113 {
114     /* define an array of integer and an integer */
115     int     arr[10];
116     int     i;

118     /* The following are examples of understanding pointers */
119 
120     // p is a pointer to an int, 
121     // or p is the address of an int
122     // Since we assume 32-bit processors, an address takes 32 bits (or 4 bytes)
123     int * p;
124 
125     // save i's address in p 
126     // Notice that i is an integer and &i is the address of the integer.
127     p = &i;
128 
129     // save arr[0]'s address in pi
130     // again arr[0] is the first element in the array, while &arr[0] is the address of the element.
131     p = &arr[0];
132 
133     // *p, p[0], arr[0] refer to the integer located at the same memory cells.
134 
135     // Let p point to the next element in the array 
136     // After the following statement, p is the address of arr[1].
137     p = p + 1;      // or p += 1;  or p ++
138 
139     // Let p point to the element after the next element 
140     // Skip two elements.
141     p += 2;
142 
143     // Now, p is the address of arr[3].
144 
145     return 0;
